void I2S3_FullReset(void)
{
    // Stop everything
    DMA1_Stream7->CR &= ~DMA_SxCR_EN;
    SPI3->I2SCFGR &= ~SPI_I2SCFGR_I2SE;
    
    // Reset I2S peripheral
     
    RCC->APB1RSTR |= RCC_APB1RSTR_SPI3RST;
    for(volatile int i = 0; i < 100; i++); // Small delay
    RCC->APB1RSTR &= ~RCC_APB1RSTR_SPI3RST;
    
    
}
